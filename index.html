// ════ 強化版：加入本地快取機制，避免重複消耗 API ════ async function preloadDynamicQuestions(artIndex) { const curArt = artworks[artIndex]; // 🏆 檢查 LocalStorage 是否已經存過這幅畫的問題 const cacheKey = `qs_${curArt.nm}`; const savedQs = localStorage.getItem(cacheKey); if (savedQs) { curArt.dynamicQs = JSON.parse(savedQs); if (curAW === artIndex) updateQuestionLabels(curArt.dynamicQs); return; } if (curArt.dynamicQs || curArt.isFetching) return; curArt.isFetching = true; const prompt = `你是一個專業的博物館語音導覽員。觀眾正在觀賞展品《${curArt.nm}》，背景：「${curArt.desc}」。請根據這段介紹，生成 ${NUM_Q} 個觀眾可能會想深入追問的問題。要求：每行一個問題，不要編號，口語化且簡短(15字內)。`; try { const response = await fetch(WORKER_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }); if (response.status === 429) throw new Error("429"); const data = await response.json(); const text = data.candidates[0].content.parts[0].text; let qs = text.split('\n').map(q => q.replace(/^[\d\.、\-*\s]+/, '').trim()).filter(q => q); while(qs.length < NUM_Q) qs.push("這幅作品還有什麼隱藏的細節嗎？"); qs = qs.slice(0, NUM_Q); // 🏆 存入本地快取，下次重新整理也不怕耗額度 localStorage.setItem(cacheKey, JSON.stringify(qs)); curArt.dynamicQs = qs; curArt.isFetching = false; if (curAW === artIndex) updateQuestionLabels(qs); } catch (err) { curArt.isFetching = false; let errMsg = err.message === "429" ? "API額度暫時用完(請等1分鐘)" : "連線不穩，點此重試"; if (curAW === artIndex) updateQuestionLabels(Array(NUM_Q).fill(errMsg)); } }
